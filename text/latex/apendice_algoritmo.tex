\refstepcounter{algorithm}%
\noindent\textbf{Algoritmo~\thealgorithm} --- \textit{Pipeline} decisório do GZ-CMD com calibração por âncoras, política de custo e trilha de auditoria.
\label{alg:gzcmd-pipeline}

\noindent\rule{\textwidth}{0.4pt}
\begin{algorithmic}[1]

\REQUIRE \parbox[t]{0.85\linewidth}{Conjunto de pares candidatos $P$ com escores brutos $\{s_i\}$;
  parâmetros $(C_{FP}, C_{FN}, C_{LLM}, e_{FP}, e_{FN})$;
  conjuntos âncora $A^+$ e $A^-$; orçamento de revisão $B$.}
\ENSURE \parbox[t]{0.85\linewidth}{Decisão $d_i \in \{\textsc{match}, \textsc{nonmatch}, \textsc{review}\}$ para cada par $i$,
  com campos de auditoria $(\mathit{razao}_i, \mathit{justificativa}_i)$.}

\medskip
\STATE \textbf{--- Fase 1: Calibração por âncoras ---}
\STATE Estimar $(\alpha,\beta)$ impondo $\mathbb{E}[\sigma(\alpha s + \beta) \mid s \in A^+] = r^+$
\STATE Estimar $(\alpha,\beta)$ impondo $\mathbb{E}[\sigma(\alpha s + \beta) \mid s \in A^-] = r^-$
\FORALL{par $i \in P$}
    \STATE $p_i \leftarrow \sigma(\alpha\, s_i + \beta)$
\ENDFOR

\medskip
\STATE \textbf{--- Fase 2: Regras de guarda determinísticas ---}
\FORALL{par $i \in P$}
    \IF{$(\mathit{data\_obito}_i - \mathit{data\_diagnostico}_i) < -180$ dias}
        \STATE $d_i \leftarrow \textsc{nonmatch}$; \quad $\mathit{razao}_i \leftarrow \texttt{FILTRO\_TEMPORAL}$
        \STATE \textbf{continuar} para o próximo par
    \ENDIF
    \IF{par $i$ compete com par $j$ pelo mesmo registro Sinan-TB}
        \STATE $\Delta p \leftarrow |p_i - p_j|$
        \IF{$\Delta p < 0{,}05$}
            \STATE Marcar ambos como candidatos a \textsc{review}; \quad $\mathit{razao}_i \leftarrow \texttt{MARGEM\_ESTREITA}$
        \ELSE
            \STATE Manter apenas o par com maior $p_i$; descartar o outro por cardinalidade
        \ENDIF
    \ENDIF
\ENDFOR

\medskip
\STATE \textbf{--- Fase 3: Motor de política baseado em perda esperada ---}
\FORALL{par $i$ ainda não resolvido}
    \STATE $L_M \leftarrow (1-p_i)\cdot C_{FP}$; \quad $L_N \leftarrow p_i\cdot C_{FN}$
    \STATE $L_R \leftarrow C_{LLM} + (1-p_i)\,e_{FP}\,C_{FP} + p_i\,e_{FN}\,C_{FN}$
    \STATE $\mathit{EVR}_i \leftarrow \min(L_M, L_N) - L_R$
    \IF{$L_M \leq L_N$ \AND $L_M \leq L_R$}
        \STATE $d_i \leftarrow \textsc{match}$
    \ELSIF{$L_N \leq L_M$ \AND $L_N \leq L_R$}
        \STATE $d_i \leftarrow \textsc{nonmatch}$
    \ELSE
        \STATE $d_i \leftarrow \textsc{review}$ \COMMENT{candidato à triagem orçamentária}
    \ENDIF
    \STATE Registrar $(L_M, L_N, L_R, \mathit{EVR}_i)$ e código de razão preliminar
\ENDFOR

\medskip
\STATE \textbf{--- Fase 4: Triagem sob restrição de orçamento ---}
\STATE Ordenar candidatos a \textsc{review} por $\mathit{EVR}_i$ em ordem decrescente
\STATE Selecionar os $k$ primeiros candidatos com $k \leq B$ para revisão assistida
\FORALL{candidato excedente com posição $> B$}
    \STATE $d_i \leftarrow \arg\min(L_M, L_N)$; \quad $\mathit{razao}_i \leftarrow \texttt{ORCAMENTO\_EXCEDIDO}$
\ENDFOR

\medskip
\STATE \textbf{--- Fase 5: Revisão assistida por LLM em protocolo dual ---}
\FORALL{par $i$ com $d_i = \textsc{review}$}
    \STATE Montar dossiê JSON com subescores, $p_{\mathit{cal}}$ e códigos de guarda
    \STATE Submeter ao Agente Objetivo e ao Agente Cético, de forma independente
    \IF{há consenso entre os dois agentes}
        \STATE Adotar decisão consensual como decisão final
    \ELSE
        \STATE Encaminhar ao Agente Árbitro para deliberação final
    \ENDIF
    \STATE Registrar código de razão, justificativa textual, versão do modelo e carimbo temporal
\ENDFOR

\medskip
\STATE \textbf{--- Fase 6: Consolidação e retorno auditável ---}
\RETURN $\{(d_i, \mathit{razao}_i, \mathit{justificativa}_i)\}$ para todo par $i \in P$

\end{algorithmic}
